524 !--------------------------------------------------------------------------
525 !!****f* types/gregorian_date
526 ! NAME
527 ! gregorian_date - Convert from a Julian day number to a Gregorian date.
528 !
529 ! SYNOPSIS
530 ! Conversion to a Gregorian calendar date from a Julian date.
531 ! Valid for any Gregorian calendar date producing a Julian day number
532 ! greater than zero.
533 !
534 ! INPUTS
535 ! iJD integer number of days that have elapsed since noon
536 ! Greenwich Mean Time (UT or TT) Monday, January 1, 4713 BC
537 ! OUTPUTS
538 ! iYear 4-digit year
539 ! iMonth 2-digit month (1-12)
540 ! iDay 2-digit day (1-31)
541 !
542 ! NOTES
543 ! Reference: Fliegel, H. F. and van Flandern, T. C. (1968).
544 ! Communications of the ACM, Vol. 11, No. 10 (October, 1968).
545 ! Modified from code found at:
546 ! http://aa.usno.navy.mil/faq/docs/JD_Formula.html
547 !
548 ! SOURCE
549 
550 elemental subroutine gregorian_date(iJD, iYear, iMonth, iDay, iOrigin)
551 
552 !! COMPUTES THE GREGORIAN CALENDAR DATE (YEAR,MONTH,DAY)
553 !! GIVEN THE JULIAN DATE (JD).
554 
555  ! [ ARGUMENTS ]
556  integer (kind=c_int), value :: iJD
557  integer (kind=c_int), intent(inout) :: iYear, iMonth, iDay
558  integer (kind=c_int), intent(in), optional :: iOrigin
559  ! [ LOCALS ]
560  integer (kind=c_int) iI,iJ,iK,iL,iN
561  integer (kind=c_int) :: iOffset
562 
563  if(present(iorigin)) then
564  ioffset = iorigin
565  else
566  ioffset = 0
567  endif
568 
569  ! allow for an alternate "origin" to be specified... technically,
570  ! this is no longer a "Julian" day, but alas... This modification
571  ! was required in order to process the "time" variables from global
572  ! climate models, which seem to be defined as something like this:
573  ! time:units = "days since 1960-01-01 00:00:00"
574  !
575  ! for the above example, JD = 2436935 on the first day; the NetCDF "time"
576  ! variable will be equal to 0. Thus, in order to get the conversion
577  ! right, we must add 0 + 2436935 to yield a true Julian Day.
578 
579  ijd = ijd + ioffset
580 
581  il= ijd + 68569_c_int
582  in= 4*il / 146097_c_int
583  il= il - (146097_c_int * in + 3_c_int)/4_c_int
584  ii= 4000_c_int * (il + 1_c_int) / 1461001_c_int
585  il= il - 1461_c_int * ii / 4_c_int + 31_c_int
586  ij= 80_c_int * il / 2447_c_int
587  ik= il - 2447_c_int * ij / 80_c_int
588  il= ij / 11_c_int
589  ij= ij + 2_c_int - 12_c_int * il
590  ii= 100_c_int * (in - 49_c_int) + ii + il
591 
592  iyear = ii
593  imonth = ij
594  iday = ik
595 
596 end subroutine gregorian_date

600 !!****f* types/julian_day
601 ! NAME
602 ! julian_day - Convert from a Gregorian calendar date to a Julian day number.
603 !
604 ! SYNOPSIS
605 ! Conversion from a Gregorian calendar date to a Julian day number.
606 ! Valid for any Gregorian calendar date producing a Julian day
607 ! greater than zero.
608 !
609 ! INPUTS
610 ! iYear 4-digit year
611 ! iMonth 2-digit month (1-12)
612 ! iDay 2-digit day (1-31)
613 !
614 ! OUTPUTS
615 ! iJD integer number of days that have elapsed since noon
616 ! Greenwich Mean Time (UT or TT) Monday, January 1, 4713 BC
617 !
618 ! SOURCE
619 
620 function julian_day ( iYear, iMonth, iDay, iOrigin ) result(iJD)
621 
622  ! [ ARGUMENTS ]
623  integer (kind=c_int), intent(in) :: iYear, iMonth, iDay
624  integer (kind=c_int), optional :: iOrigin
625 
626  ! [ LOCALS ]
627  integer (kind=c_int) i,j,k
628  integer (kind=c_int) :: iOffset
629  character (len=256) :: sBuf
630 
631  ! [ RETURN VALUE ]
632  integer (kind=c_int) :: iJD
633 
634  i= iyear
635  j= imonth
636  k= iday
637 
638  if(.not. (imonth >= 1 .and. imonth <= 12)) then
639  write(sbuf,fmt="('Illegal month value given: ',i4)") imonth
640  call assert( lfalse, trim(sbuf), __srcname__, __line__)
641  elseif(.not. (iday >= 1 .and. iday <= 31)) then
642  write(sbuf,fmt="('Illegal day value given: ',i4)") iday
643  call assert( lfalse, trim(sbuf), __srcname__, __line__)
644  endif
645 
646  if(present(iorigin)) then
647  ioffset = iorigin
648  else
649  ioffset = 0
650  endif
651 
652  ijd= ( k-32075_c_int + 1461_c_int * (i + 4800_c_int + (j - 14_c_int) / 12_c_int) &
653  /4_c_int + 367_c_int * (j - 2_c_int - (j - 14_c_int)/ 12_c_int * 12_c_int) &
654  /12_c_int - 3_c_int *((i + 4900_c_int + (j - 14_c_int) &
655  /12_c_int)/100_c_int)/4_c_int ) - ioffset
656 
657 end function julian_day
