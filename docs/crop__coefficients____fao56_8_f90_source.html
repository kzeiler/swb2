393  impure elemental function update_crop_coefficient_date_as_threshold( iLanduseIndex ) &
394  result(kcb)
395 
396  integer (c_int), intent(in) :: ilanduseindex
397  real (c_float) :: kcb

399  ! [ LOCALS ]

401 
402  if ( kcb_method( ilanduseindex ) == kcb_method_monthly_values ) then

404  kcb = kcb_l( sim_dt%curr%iMonth, ilanduseindex )

406  else
407 
408  ! define shorthand variable names for remainder of function
409  associate( date_ini => growth_stage_date( enddate_ini, ilanduseindex ), &
410  date_dev => growth_stage_date( enddate_dev, ilanduseindex ), &
411  date_mid => growth_stage_date( enddate_mid, ilanduseindex ), &
412  date_late => growth_stage_date( enddate_late, ilanduseindex ), &
413  date_fallow => growth_stage_date( enddate_fallow, ilanduseindex ), &
414  kcb_ini => kcb_l(kcb_ini, ilanduseindex), &
415  kcb_mid => kcb_l(kcb_mid, ilanduseindex), &
416  kcb_min => kcb_l(kcb_min, ilanduseindex), &
417  plantingdate => growth_stage_date( planting_date, ilanduseindex), &
418  kcb_end => kcb_l(kcb_end, ilanduseindex), &
419  current_date => sim_dt%curr )
420 
421  ! now calculate Kcb for the given landuse
422 
423  if( current_date > date_late ) then
424 
425  kcb = kcb_min
426 
427  elseif ( current_date > date_mid ) then
428 
429  ffrac = ( current_date - date_mid ) / ( date_late - date_mid )
430 
431  kcb = kcb_mid * (1.0_c_double - ffrac) + kcb_end * ffrac
432 
433  elseif ( current_date > date_dev ) then
434 
435  kcb = kcb_mid
436 
437  elseif ( current_date > date_ini ) then
438 
439  ffrac = ( current_date - date_ini ) / ( date_dev - date_ini )
440 
441  kcb = kcb_ini * (1.0_c_double - ffrac) + kcb_mid * ffrac
442 
443  elseif ( current_date >= plantingdate ) then
444 
445  kcb = kcb_ini
446 
447  else
448 
449  kcb = kcb_min
450 
451  endif
452 
453  end associate
454 
455  end if
