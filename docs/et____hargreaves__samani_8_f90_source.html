82elemental function et_hargreaves_calculate( iDayOfYear, iNumDaysInYear, fLatitude, fTMin, fTMax ) result(fReferenceET0)
83 !! Computes the potential ET for each cell, based on TMIN and TMAX.
84 !! Stores cell-by-cell PET values in the model grid.
85
86 integer (c_int),intent(in) :: idayofyear
87 integer (c_int),intent(in) :: inumdaysinyear
88 real (c_float), intent(in) :: flatitude
89 real (c_float), intent(in) :: ftmin
90 real (c_float), intent(in) :: ftmax
91 real (c_float) :: freferenceet0
92
93 ! [ LOCALS ]
94 real (c_double) :: fdelta, fomega_s, fd_r, fra
95 real (c_double) :: dlatitude_radians
96
97 dlatitude_radians = flatitude * degrees_to_radians
98
99 fd_r =relative_earth_sun_distance__d_r(idayofyear,inumdaysinyear)
100 fdelta = solar_declination_simple__delta(idayofyear, inumdaysinyear)
101
102 fomega_s = sunrise_sunset_angle__omega_s(dlatitude_radians, fdelta)
103
104 ! NOTE that the following equation returns extraterrestrial radiation in
105 ! MJ / m**2 / day. The Hargreaves equation requires extraterrestrial
106 ! radiation to be expressed in units of mm / day.
107 fra = extraterrestrial_radiation__ra(dlatitude_radians, fdelta, fomega_s, fd_r)
108
109 freferenceet0 = et0_hargreaves( equivalent_evaporation(fra), ftmin, ftmax)
110
111end function et_hargreaves_calculate

128 rtavg = (rtminf + rtmaxf) / 2.0_c_double

138! * (F_to_C(rTavg) + pConfig%fET_Constant) &
139! * (rTDelta**fET_Exponent)) &

142end function et0_hargreaves
