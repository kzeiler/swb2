532 integer (c_int) :: iStat
533 character (len=:), allocatable :: description_str
534 character (len=:), allocatable :: output_filesname_str
535 character (len=256), save :: previous_zone_filename
536
537 ! OK, if this is the same ARC ASCII grid file processed in a previous iteration,
538 ! we leave the values as they are and skip reading in the grid again
539 if ( .not. trim(previous_zone_filename)==trim(grid_filename) ) then
540
541 ! remove file extension and path from zone filename
542 description_str = left( grid_filename, substring=".")
543 if (description_str .contains. "/") description_str = right( description_str, substring="/")
544 if (description_str .contains. "\") description_str = right( description_str, substring="\")
545
546 output_filesname_str = "zone_grid__"//trim(description_str)//"__as_read_into_SWBSTATS2.asc"
547
548 ! allocate memory for a generic data_catalog_entry
549 if (associated(this%grd_zone)) deallocate(this%grd_zone)
550 nullify(this%grd_zone)
551 allocate(this%grd_zone)
552
553 call this%grd_zone%set_target_PROJ4(this%target_proj4_string)
554 call this%grd_zone%initialize( &
555 sdescription="Zone Grid", &
556 sfiletype="ARC_GRID", &
557 sfilename=trim(grid_filename), &
558 idatatype=datatype_int )
559
560 call this%grd_zone%getvalues()
561
562! where ( this%grd_native%dpData <= NC_FILL_FLOAT )
563! this%grd_zone%pGrdBase%iData = NC_FILL_INT
564! end where
565
566 call grid_writearcgrid(output_filesname_str, this%grd_zone%pGrdBase )
567
568 ! set value of previous filename to the name of the grid we just processed.
569 previous_zone_filename = trim(grid_filename)
570
571 call this%unique_zone_list%clear()
572 call this%get_unique_int(this%grd_zone%pGrdBase%iData, this%unique_zone_list)
573
574 endif

861 subroutine calculate_slice_statistics( this, grid_delta, grid_delta2, start_date, end_date, grid_mask )

877 ! force slice dates to honor bounds of data dates
878 if ( start_date < this%data_start_date ) start_date = this%data_start_date

889
890 allocate(local_mask(ubound(grd_sum,1),ubound(grd_sum,2)))
891
892 day_count = 0

894 ! SIM_DT is being used here simply because it provides a simple way to march through

902

913 if ( netcdf_update_time_starting_index(this%ncfile_in, julian_day_number ) ) then

931

933 grd_mean = 0.0_c_double
934 grd_var = 0.0_c_double
935
936 endif
937
938 where ( local_mask )
939 delta = grd_new - grd_mean
940 grd_sum = grd_sum + grd_new
941 grd_mean = grd_mean + delta / real( day_count, c_double )
942 delta2 = grd_new - grd_mean
943 grd_var = grd_var + delta * delta2
944 end where
945
946 ! write(*,fmt="(a,f14.3,f14.3)") "gridsum: ", minval(grd_sum, mask=grd_new > NC_FILL_FLOAT), &
947 ! maxval(grd_sum, mask=grd_new > NC_FILL_FLOAT)

954
955 call sim_dt%addDay
956 if ( sim_dt%curr > end_date ) exit
957
958 end do
959
960 where ( .not. local_mask )
961 grd_mean = nc_fill_float
962 grd_sum = nc_fill_float
963 grd_var = nc_fill_float
964 ! elsewhere
965 ! grd_var = grd_var / (counter - 1)
966 endwhere
967

971! grd_var = grd_var/ real( day_count, c_float ) * 365.25

974 ! else

977 ! end where
