5 ! Subtract-and-borrow random number generator proposed by
6 ! Marsaglia and Zaman, implemented by F. James with the name
7 ! RCARRY in 1991, and later improved by Martin Luescher
8 ! in 1993 to produce "Luxury Pseudorandom Numbers".
9 ! Fortran 77 coded by F. James, 1993
10 
11 ! References:
12 ! M. Luscher, Computer Physics Communications 79 (1994) 100
13 ! F. James, Computer Physics Communications 79 (1994) 111
14 
15 ! This Fortran 90 version is by Alan Miller (alan @ mel.dms.csiro.au)
16 ! Latest revision - 11 September 1995

18 ! LUXURY LEVELS.
19 ! ------ ------ The available luxury levels are:
20 
21 ! level 0 (p=24): equivalent to the original RCARRY of Marsaglia
22 ! and Zaman, very long period, but fails many tests.
23 ! level 1 (p=48): considerable improvement in quality over level 0,
24 ! now passes the gap test, but still fails spectral test.
25 ! level 2 (p=97): passes all known tests, but theoretically still
26 ! defective.
27 ! level 3 (p=223): DEFAULT VALUE. Any theoretically possible
28 ! correlations have very small chance of being observed.
29 ! level 4 (p=389): highest possible luxury, all 24 bits chaotic.
30 
31 !!!! ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
32 !!!! Calling sequences for RANLUX: ++
33 !!!! CALL RANLUX (RVEC, LEN) returns a vector RVEC of LEN ++
34 !!!! 32-bit random floating point numbers between ++
35 !!!! zero (not included) and one (also not incl.). ++
36 !!!! CALL RLUXGO(LUX,INT,K1,K2) initializes the generator from ++
37 !!!! one 32-bit integer INT and sets Luxury Level LUX ++
38 !!!! which is integer between zero and MAXLEV, or if ++
39 !!!! LUX .GT. 24, it sets p=LUX directly. K1 and K2 ++
40 !!!! should be set to zero unless restarting at a break++
41 !!!! point given by output of RLUXAT (see RLUXAT). ++
42 !!!! CALL RLUXAT(LUX,INT,K1,K2) gets the values of four integers++
43 !!!! which can be used to restart the RANLUX generator ++
44 !!!! at the current point by calling RLUXGO. K1 and K2++
45 !!!! specify how many numbers were generated since the ++
46 !!!! initialization with LUX and INT. The restarting ++
47 !!!! skips over K1+K2*E9 numbers, so it can be long.++
48 !!!! A more efficient but less convenient way of restarting is by: ++
49 !!!! CALL RLUXIN(ISVEC) restarts the generator from vector ++
50 !!!! ISVEC of 25 32-bit integers (see RLUXUT) ++
51 !!!! CALL RLUXUT(ISVEC) outputs the current values of the 25 ++
52 !!!! 32-bit integer seeds, to be used for restarting ++
53 !!!! ISVEC must be dimensioned 25 in the calling program ++
54 !!!! ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

84 SUBROUTINE ranlux(rvec, lenv)
85 
86 IMPLICIT NONE
87 
88 integer (c_int), INTENT(IN) :: lenv
89 real(c_float), INTENT(OUT) :: rvec(lenv)
90 
91 ! Local variables
92 
93 integer (c_int) :: i, k, lp, ivec, iseeds(24), isk
94 real(c_float) :: uni
95 
96 ! NOTYET is .TRUE. if no initialization has been performed yet.
97 ! Default Initialization by Multiplicative Congruential
98 
99 IF (notyet) THEN
100  notyet = .false.
101  jseed = jsdflt
102  inseed = jseed
103  WRITE (6,'(A,I12)') ' RANLUX DEFAULT INITIALIZATION: ', jseed
104  luxlev = lxdflt
105  nskip = ndskip(luxlev)
106  lp = nskip + 24
107  in24 = 0
108  kount = 0
109  mkount = 0
110  WRITE (6,'(A,I2,A,I4)') ' RANLUX DEFAULT LUXURY LEVEL = ', luxlev, &
111  ' p =', lp
112  twom24 = 1.
113  DO i = 1, 24
114  twom24 = twom24 * 0.5
115  k = jseed / 53668
116  jseed = 40014 * (jseed-k*53668) - k * 12211
117  IF (jseed.LT.0) jseed = jseed + icons
118  iseeds(i) = mod(jseed,itwo24)
119  END DO
120  twom12 = twom24 * 4096.
121  DO i = 1, 24
122  seeds(i) = real(c_float)(iseeds(i)) * twom24
123  next(i) = i - 1
124  END DO
125  next(1) = 24
126  i24 = 24
127  j24 = 10
128  carry = 0.
129  IF (seeds(24).EQ.0.) carry = twom24
130 END IF
131 
132 ! The Generator proper: "Subtract-with-borrow",
133 ! as proposed by Marsaglia and Zaman,
134 ! Florida State University, March, 1989
135 
136 DO ivec = 1, lenv
137  uni = seeds(j24) - seeds(i24) - carry
138  IF (uni.LT.0.) THEN
139  uni = uni + 1.0
140  carry = twom24
141  ELSE
142  carry = 0.
143  END IF
144  seeds(i24) = uni
145  i24 = next(i24)
146  j24 = next(j24)
147  rvec(ivec) = uni
148 ! small numbers (with less than 12 "significant" bits) are "padded".
149  IF (uni.LT.twom12) THEN
150  rvec(ivec) = rvec(ivec) + twom24 * seeds(j24)
151 ! and zero is forbidden in case someone takes a logarithm
152  IF (rvec(ivec).EQ.0.) rvec(ivec) = twom24 * twom24
153  END IF
154 ! Skipping to luxury. As proposed by Martin Luscher.
155  in24 = in24 + 1
156  IF (in24.EQ.24) THEN
157  in24 = 0
158  kount = kount + nskip
159  DO isk = 1, nskip
160  uni = seeds(j24) - seeds(i24) - carry
161  IF (uni.LT.0.) THEN
162  uni = uni + 1.0
163  carry = twom24
164  ELSE
165  carry = 0.
166  END IF
167  seeds(i24) = uni
168  i24 = next(i24)
169  j24 = next(j24)
170  END DO
171  END IF
172 END DO
173 kount = kount + lenv
174 IF (kount.GE.igiga) THEN
175  mkount = mkount + 1
176  kount = kount - igiga
177 END IF
178 RETURN
179 
180 END SUBROUTINE ranlux

275 SUBROUTINE rluxgo(lux, ins, k1, k2)
276 
277 IMPLICIT NONE

280 
281 ! Local variables
282

284 real(c_float) :: uni
285 
286 IF (lux.LT.0) THEN
287  luxlev = lxdflt
288 ELSE IF (lux.LE.maxlev) THEN
289  luxlev = lux
290 ELSE IF (lux.LT.24.OR.lux.GT.2000) THEN

293 ELSE

295  DO ilx = 0, maxlev
296  IF (lux.EQ.ndskip(ilx)+24) luxlev = ilx
297  END DO
298 END IF

301  WRITE (6,'(A,I2,A,I4)') ' RANLUX LUXURY LEVEL SET BY RLUXGO :', luxlev, &
302  ' P=', nskip + 24
303 ELSE
304  nskip = luxlev - 24
305  WRITE (6,'(A,I5)') ' RANLUX P-VALUE SET BY RLUXGO TO:', luxlev
306 END IF

308 IF (ins.LT.0) WRITE (6,'(A)') &
309  ' Illegal initialization by RLUXGO, negative input seed'
310 IF (ins.GT.0) THEN

312  WRITE (6,'(A,3I12)') ' RANLUX INITIALIZED BY RLUXGO FROM SEEDS', jseed, k1, k2
313 ELSE

315  WRITE (6,'(A)') ' RANLUX INITIALIZED BY RLUXGO FROM DEFAULT SEED'
316 END IF

318 notyet = .false.
319 twom24 = 1.
320 DO i = 1, 24
321  twom24 = twom24 * 0.5
322  k = jseed / 53668
323  jseed = 40014 * (jseed-k*53668) - k * 12211
324  IF (jseed.LT.0) jseed = jseed + icons
325  iseeds(i) = mod(jseed,itwo24)
326 END DO
327 twom12 = twom24 * 4096.
328 DO i = 1, 24
329  seeds(i) = real(c_float)(iseeds(i)) * twom24
330  next(i) = i - 1
331 END DO
332 next(1) = 24
333 i24 = 24
334 j24 = 10
335 carry = 0.
336 IF (seeds(24).EQ.0.) carry = twom24
337 ! If restarting at a break point, skip K1 + IGIGA*K2
338 ! Note that this is the number of numbers delivered to
339 ! the user PLUS the number skipped (if luxury .GT. 0).
340 kount = k1
341 mkount = k2
342 IF (k1+k2.NE.0) THEN
343  DO iouter = 1, k2 + 1
344  inner = igiga
345  IF (iouter.EQ.k2+1) inner = k1
346  DO isk = 1, inner
347  uni = seeds(j24) - seeds(i24) - carry
348  IF (uni.LT.0.) THEN
349  uni = uni + 1.0
350  carry = twom24
351  ELSE
352  carry = 0.
353  END IF
354  seeds(i24) = uni
355  i24 = next(i24)
356  j24 = next(j24)
357  END DO
358  END DO
359 ! Get the right value of IN24 by direct calculation
360  in24 = mod(kount,nskip+24)
361  IF (mkount.GT.0) THEN
362  izip = mod(igiga, nskip+24)
363  izip2 = mkount * izip + in24
364  in24 = mod(izip2, nskip+24)
365  END IF
366 ! Now IN24 had better be between zero and 23 inclusive
367  IF (in24.GT.23) THEN
368  WRITE (6,'(A/A,3I11,A,I5)') &
369  ' Error in RESTARTING with RLUXGO:', ' The values', ins, &
370  k1, k2, ' cannot occur at luxury level', luxlev
371  in24 = 0
372  END IF
373 END IF
374 RETURN
